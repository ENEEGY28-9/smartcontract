// Test function for any Solana public key
async function testCustomKey(publicKeyString) {
    try {
        console.log(`ğŸ” Testing: ${publicKeyString.slice(0, 8)}...${publicKeyString.slice(-8)}`);

        // Validate format
        if (publicKeyString.length < 32 || publicKeyString.length > 44) {
            throw new Error('Invalid length. Solana addresses are 32-44 characters.');
        }

        const base58Regex = /^[1-9A-HJ-NP-Za-km-z]+$/;
        if (!base58Regex.test(publicKeyString)) {
            throw new Error('Invalid format. Use alphanumeric characters only.');
        }

        // Test with blockchain
        const connection = new solanaWeb3.Connection('https://api.devnet.solana.com');
        const publicKey = new solanaWeb3.PublicKey(publicKeyString);

        const balanceLamports = await connection.getBalance(publicKey);
        const balance = balanceLamports / 1_000_000_000;

        const accountInfo = await connection.getAccountInfo(publicKey);
        const exists = accountInfo !== null;

        console.log('âœ… VALID PUBLIC KEY');

        console.log(`ğŸ“Š Account exists: ${exists}`);

        return { valid: true, balance, exists };

    } catch (error) {
        console.error(`âŒ INVALID: ${error.message}`);
        return { valid: false, error: error.message };
    }
}
