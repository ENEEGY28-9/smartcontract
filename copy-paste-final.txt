// Test function for any public key
async function testWalletKey(publicKeyString) {
    try {
        console.log(`ðŸ” Testing: ${publicKeyString.slice(0, 8)}...`);

        // Validate and get info
        const connection = new solanaWeb3.Connection('https://api.devnet.solana.com');
        const publicKey = new solanaWeb3.PublicKey(publicKeyString);

        const balance = await connection.getBalance(publicKey);
        const balanceInSOL = balance / solanaWeb3.LAMPORTS_PER_SOL;

        const accountInfo = await connection.getAccountInfo(publicKey);
        const exists = accountInfo !== null;

        console.log(`âœ… Valid Public Key`);
        console.log(`ðŸ’° Balance: ${balanceInSOL.toFixed(6)} SOL`);
        console.log(`ðŸ“Š Account Exists: ${exists}`);
        console.log(`ðŸŒ Network: Solana Devnet`);

        return { valid: true, balance: balanceInSOL, exists, address: publicKeyString };

    } catch (error) {
        console.error(`âŒ Invalid: ${error.message}`);
        return { valid: false, error: error.message };
    }
}
